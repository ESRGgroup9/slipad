119void *CCommunication::tSend(void *arg)
120{
121 // get CCommunication instance
122 CCommunication *ccomm = reinterpret_cast<CCommunication*>(arg);
123 string msg;
124
125 do
126 {
127 pthread_mutex_lock(&ccomm->mutTxMsgs);
128
129 // is there any message queued to send?
130 if(ccomm->TxMsgs.empty())
131 {
132 // no messages to send. wait for condtSend
133 DEBUG_MSG("[CComms::tSend] Waiting for condtSend...");
134 pthread_cond_wait(&ccomm->condtSend, &ccomm->mutTxMsgs);
135
136 if(ccomm->status == ConnStatus::CLOSED)
137 break;
138
139 DEBUG_MSG("[CComms::tSend] Im awake!");
140 }
141
142 // pop msg from queue
143 msg = ccomm->TxMsgs.front();
144 // Removes the next element in the queue, reducing its size by one
145 ccomm->TxMsgs.pop();
146 // ccomm->TxMsgs.pop_front();
147 pthread_mutex_unlock(&ccomm->mutTxMsgs);
148
149 DEBUG_MSG("[CComms::tSend] Popped(" << msg << ") - " << ccomm->TxMsgs.size() << " msgs queued");
150
151 // send message
152 int ret = -1;
153 int err = EAGAIN;
154 do
155 {
156 ret = ccomm->send(msg);
157 err = errno;
158 }
159 // wait for non blocking operation
160 while((ret == -1) && (err == EAGAIN));
161
162 DEBUG_MSG("[CComms::tSend] Sent(" << msg << ")");
163 msg.clear();
164 }
165 while(ccomm->status != ConnStatus::CLOSED);
166
167 return NULL;
168}
