111void CSensors::run()
112{
113 char msg[MAX_MSG_LEN_R];
114 int err = 0;
115
116 // clear message
117 memset(msg, 0, sizeof(msg));
118
119 DEBUG_MSG("[CSensors::run] Waiting for main PID...");
120 // receive main PID
121 do
122 {
123 // read message from msgqueue
124 err = mq_receive(msgqSensors, msg, MAX_MSG_LEN_R, NULL);
125 if(err == -1)
126 {
127 // get error from errno
128 err = errno;
129
130 // error not expected
131 if(err != EAGAIN)
132 panic("In mq_receive()");
133 }
134 // while message queue is empty = wait for PID receival
135 } while(err == EAGAIN);
136
137 // mainPID received
138 mainPID = static_cast<int>(atoi(msg));
139 DEBUG_MSG("[CSensors::run] Received main PID[" << mainPID << "]");
140
141#ifdef DEBUG
142 lampf.enable();
143 pir.enable();
144#endif // !DEBUG
145
146 // start sampling LDR sensor
147 timReadLdr.start();
148
149 // wait for thread termination
150 pthread_join(tReadLdr_id, NULL);
151}

153void *CSensors::tReadLdr(void *arg)
154{
155 CSensors *c = reinterpret_cast<CSensors*>(arg);
156 LuxState luxState = LuxState::DAY;
157 LuxState oldLuxState = LuxState::DAY;
158
159 while(c)
160 {
161 pthread_mutex_lock(&c->mutReadLdr);
162
163 DEBUG_MSG("[CSensors::tReadLdr] Waiting for condReadLdr...");
164 pthread_cond_wait(&c->condReadLdr, &c->mutReadLdr);
165 DEBUG_MSG("[CSensors::tReadLdr] Im awake!");
166
167 luxState = c->ldr.getLuxState();
168 pthread_mutex_unlock(&c->mutReadLdr);
169
170 // is there a change in lux State?
171 if(luxState != oldLuxState)
172 {
173 if(luxState == LuxState::NIGHT)
174 {
175 c->lampf.enable();
176 c->pir.enable();
177 c->sendCmd("MIN");
178 }
179 else
180 {
181 c->lampf.disable();
182 c->pir.disable();
183 c->sendCmd("OFF");
184 }
185
186 oldLuxState = luxState;
187 }
188 }
189
190 DEBUG_MSG("[CSensors::tReadLdr] Exiting thread...");
191 return NULL;
192}
