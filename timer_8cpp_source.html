84void Timer::setPeriod(unsigned period_secs)
85{
86 // period between now and the first timer interrupt
87 ts.it_value.tv_sec = period_secs;
88 ts.it_value.tv_nsec = 0;
89 // period between successive timer interrupts
90 ts.it_interval.tv_sec = period_secs * (is_periodic);
91 ts.it_interval.tv_nsec = 0;
92
93 if (timer_settime(timer_id, 0, &ts, 0) == -1)
94 panic("Set timer");
95}
