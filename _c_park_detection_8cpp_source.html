30bool CParkDetection::getOutline(void)
31{
32 Mat frame, gray;
33
34 frame = imread(IMAGE_PATH, IMREAD_COLOR);
35 if( frame.empty() )
36 {
37 panic("Couldn't load image");
38 }
39
40 Mat gray0(frame.size(), CV_8U);
41
42 // Filter the noise
43 Mat gaussian;
44 GaussianBlur(frame, gaussian, Size(7, 7), 0);
45
46 vector<vector<Point> > contours;
47
48 // empty the vector
49 parkCoords.clear();
50
51 // find squares in every color plane of the image
52 for( int c = 0; c < 3; c++ )
53 {
54 int ch[] = {c, 0};
55
56 mixChannels(&gaussian, 1, &gray0, 1, ch, 1);
57 // try several threshold levels
58 for( int l = 0; l < levels; l++ )
59 {
60 // Canny helps to catch squares with gradient shading
61 if( l == 0 )
62 {
63 // apply Canny. Take the upper threshold from slider
64 // and set the lower to 0 (which forces edges merging)
65 Canny(gray0, gray, 0, thresh, 5);
66
67 // dilate canny output to remove potential
68 // holes between edge segments
69 dilate(gray, gray, Mat(), Point(-1,-1));
70 }
71 else
72 {
73 // apply threshold if l!=0:
74 // tgray(x,y) = gray(x,y) < (l+1)*255/N ? 255 : 0
75 gray = gray0 >= (l+1)*255/levels;
76 }
77
78 // find contours and store them all as a list
79 findContours(gray, contours, RETR_LIST, CHAIN_APPROX_SIMPLE);
80
81 // test each contour
82 for( size_t i = 0; i < contours.size(); i++ )
83 {
84 addPark(contours[i]);
85 }
86 }
87 }
88
89 vacantsNum = parkCoords.size();
90
91 if(vacantsNum == 0)
92 return false;
93
94 return true;
95}

98{

145vector<Rect> CParkDetection::detectCars(const Mat& frame)
146{
147 Mat gray;
148 cvtColor(frame, gray, COLOR_BGR2GRAY);
149
150 // Load a classifier from its XML description
151 CascadeClassifier classifier("/etc/slipad/haar_cars1.xml");
152
153 if( classifier.empty() )
154 DEBUG_MSG("Classifier not loaded.\n");
155
156 Mat gaussian;
157 // blur the image
158 GaussianBlur(gray, gaussian, Size(5, 5), 0);
159
160 Mat ones = Mat(3, 3, CV_8UC1, Scalar(1));
161 Mat dilated;
162 // dilate canny output to remove potential
163 // holes between edge
164 dilate( gaussian, dilated, ones );
165
166 // get the desired kernel
167 Mat kernel = getStructuringElement(MORPH_ELLIPSE, Size(2, 2));
168
169 Mat output;
170 // apply morphology transformation with the desired kernel
171 morphologyEx(dilated, output, MORPH_CLOSE, kernel);
172
173 // Prepare a vector where the detected features will be stored
174 vector<Rect> features;
175
176 // Detect the features in the normalized, gray-scale version of the image
177 classifier.detectMultiScale(output, features, 1.1, 2,
178 0 | CASCADE_SCALE_IMAGE, Size(50, 50));
179
180 return features;

183void CParkDetection::addPark( vector<Point>& contour )
184{
185 vector<Point> approx;
186
187 // aproximate the contour to a polygon
188 approxPolyDP(contour, approx, arcLength(contour, true)*0.02, true);
189
190 // A square should have 4 vertices after aproximation &&
191 // have a relatively large area &&
192 // have a not so big area &&
193 // be convex
194 if( (approx.size() == 4) &&
195 (fabs(contourArea(approx)) > 3000) &&
196 (fabs(contourArea(approx)) < 20000) &&
197 isContourConvex(approx) )
198 {
199 double maxCosine = 0;
200 // find the maximum cosine of all angles of the polygon
201 for( int j = 2; j < 5; j++ )
202 {
203 double cosine = fabs(angle_cosine(approx[j%4], approx[j-2], approx[j-1]));
204 maxCosine = MAX(maxCosine, cosine);
205 }
206
207 // to be consired a rectangle it must have small cosines (~90 degrees)
208 // && not be overlapped with other rectangles
209 if( maxCosine < 0.1 && (isOverlapp(approx) == -1) )
210 {
211 parkStatus.push_back(1); // all available
212 parkCoords.push_back(approx);
213 vacantsNum++;
214 }
215 }

218int CParkDetection::isOverlapp(vector<Point>& approx)
219{
220 Point approxCenter = findCenter(approx);
221
222 for( int i=0; i < parkCoords.size(); i++ )
223 {
224 Point parkCoordsCenter = findCenter(parkCoords[i]);
225
226 // centers are overlapped? Distance between centers inferior to 30
227 if( fabs(parkCoordsCenter.x - (approxCenter.x)) < 50 &&
228 fabs(parkCoordsCenter.y - (approxCenter.y)) < 50 )
229 return i;
230 }
231
232 return -1;

243double CParkDetection::angle_cosine(const Point pt1, const Point pt2, const Point pt0 )
244{
245 double dx1 = pt1.x - pt0.x;
246 double dy1 = pt1.y - pt0.y;
247 double dx2 = pt2.x - pt0.x;
248 double dy2 = pt2.y - pt0.y;
249
250 return (dx1*dx2 + dy1*dy2)/sqrt((dx1*dx1 + dy1*dy1)*(dx2*dx2 + dy2*dy2) + 1e-10);
251}
