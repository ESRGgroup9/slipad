78void CRemoteSystem::timCheckConnISR()
79{
80 int size = clientList.size();
81 int i = 0;
82
83 while(i < size)
84 {
85 if(clientList[i]->info.state == ConnStatus::CLOSED)
86 {
87 // client has disconnected
88 DEBUG_MSG("[CRemoteSystem::checkConn] Removing dead client with sockfd[" << clientList[i]->info.sockfd << "]");
89 // remove it from the client list
90 clientList.erase(clientList.begin() + i);
91 // decrease client list size used as loop variable
92 size--;
93 // decrease number of connected clients to the server
94 server.numClients--;
95
96 /* do not increment i
97 since 'erase' deletes the position, in the next iteration the same
98 value for 'i' will index the next vector element*/
99 }
100 else
101 // go to the next element
102 i++;
103 }
104}

125void CRemoteSystem::run()
126{
127 int sd;
128
129 DEBUG_MSG("[CRemoteSystem::run] Listening for new connections...");
130 timCheckConn.start();
131
132 while(1)
133 {
134 sd = server.accept();
135
136 if(sd != -1)
137 {
138 pthread_t recvType_id;
139 // handle client addition to client list in detached thread
140 client_port = sd;
141 pthread_create(&recvType_id, NULL, tRecvType, this);
142 pthread_detach(recvType_id);
143 DEBUG_MSG("[CRemoteSystem::run] Continue listening for new connections...");

148int CRemoteSystem::typeCb(int argc, char *argv[])
149{
150 if(argc != 2)
151 {
152 DEBUG_MSG("[CRemoteSystem::typeCb] Usage: TYPE;<clientType>");
153 return -1;
154 }
155
156 ClientType type = static_cast<ClientType>(atoi(argv[1]));
157 CRemoteClient *client = NULL;
158
159 switch(type)
160 {
161 case ClientType::GATEWAY:
162 client = new RCGateway(thisPtr->client_port, thisPtr->db);
163 break;
164
165 case ClientType::WEBSITE:
166 client = new RCWebsite(thisPtr->client_port, thisPtr->db);
167 break;
168
169 case ClientType::APPLICATION:
170 client = new RCApplication(thisPtr->client_port, thisPtr->db);
171 break;
172
173 default:
174 // do not handle other types - client stays NULL
175 break;
176 }
177
178 if(client == NULL)
179 {
180 DEBUG_MSG("[CRemoteSystem::typeCb] Client not created");
181 return -1;
182 }
183
184 pthread_mutex_lock(&thisPtr->mutAddClient);
185 // add new client
186 thisPtr->clientList.push_back(client);
187 // execute respective init and run methods
188 int i = thisPtr->clientList.size() - 1;
189 pthread_mutex_unlock(&thisPtr->mutAddClient);
190
191 thisPtr->clientList[i]->init(1,2);
192 thisPtr->clientList[i]->run();
193 DEBUG_MSG("[CRemoteSystem::typeCb] Client of type(" << static_cast<int>(type) << ") created successfully");
194
195 return 0;
196}

198void *CRemoteSystem::tRecvType(void *arg)
199{
200 // get CRemoteSystem instance
201 CRemoteSystem *c = reinterpret_cast<CRemoteSystem*>(arg);
202 CTCPComm tcp(c->client_port);
203
204 string msg;
205 int ret = 0;
206 int err = -1;
207
208 do
209 {
210 ret = tcp.recv(msg);
211
212 if(ret == 0)
213 // client has closed the connection
214 return NULL;
215
216 if(ret > 0)
217 {
218 // parse received string
219 err = c->typeParser.parse(msg.c_str());
220 }
221 }
222 while(err != 0);
223
224 return NULL;
225}
