96 tcp->init(1);

98 // wait for threads termination
99 pthread_join(tTCPRecv_id, NULL);
100

102 tcp->run();

105 tcp->~CTCPclient();

107 // Since a socket that had been connected once cannot be reused with
108 // another call to connect() one has to create TCPclient dynamically

114void *CGateway::tTCPRecv(void *arg)
115{
116 // get CGateway instance
117 CGateway *c = reinterpret_cast<CGateway*>(arg);
118 string msg;
119 int ret = -1;
120
121 do
122 {
123 ret = c->tcp->recv(msg);
124
125 if(ret == -1)
126 {
127 int err = errno;
128
129 // non blocking operation?
130 if(err != EAGAIN)
131 {
132 // unexpected error
133 stringstream debug_msg;
134 debug_msg << "ERROR> [CGateway::tTCPRecv] " << string(strerror(err)) << endl;
135 ERROR_MSG(debug_msg.str().c_str());
136 }
137 }
138 else if(ret > 0)
139 {
140 // recv has returned the number of bytes read
141 // "parse" receive message. Get message payload and destination addr
142 size_t i = msg.find_last_of(PARSE_DELIMITER);
143 // set message payload
144 string str = msg.substr(0,i);
145 // set destination - receiver address
146 int destAddr = atoi(msg.substr(i+1).c_str());
147
148 c->lora.setDestAddr(destAddr);
149 c->lora.push(str);
150
151 stringstream debug_msg;
152 debug_msg << "[CGateway::tTCPRecv] Forwarding[" << str << "] to LS[" << destAddr << "]" << endl;
153 DEBUG_MSG(debug_msg.str().c_str());
154 }
155 }
156 while(ret != 0);
157
158 DEBUG_MSG("[CGateway::tTCPRecv] RS has closed the connection");
159 return NULL;
160}
