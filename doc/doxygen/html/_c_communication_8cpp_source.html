77void *CCommunication::tSend(void *arg)
78{
79 // get CCommunication instance
80 CCommunication *ccomm = reinterpret_cast<CCommunication*>(arg);
81 string msg;
82
83 DEBUG_MSG("[CComms::tSend] entering thread");
84
85 while(ccomm)
86 {
87 pthread_mutex_lock(&ccomm->mutTxMsgs);
88
89 // is there any message queued to send?
90 if(ccomm->TxMsgs.empty())
91 {
92 // no messages to send. wait for condtSend
93 DEBUG_MSG("[CComms::tSend] Waiting for condtSend...");
94 pthread_cond_wait(&ccomm->condtSend, &ccomm->mutTxMsgs);
95 DEBUG_MSG("[CComms::tSend] Im awake!");
96 }
97
98 // pop msg from queue
99 msg = ccomm->TxMsgs.front();
100 // Removes the next element in the queue, reducing its size by one
101 ccomm->TxMsgs.pop();
102 pthread_mutex_unlock(&ccomm->mutTxMsgs);
103
104 DEBUG_MSG("[CComms::tSend] Popped(" << msg << ") - [" << ccomm->TxMsgs.size() << "] msgs queued");
105 // send message
106 ccomm->send(msg);
107 cout << "[CComms::tSend] Sent(" << msg << ")" << endl;
108 msg.clear();
109 }
110
111 return NULL;
112}
