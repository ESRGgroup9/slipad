185 private static function processToken(int $state, string $token)
186 {
187 switch ($state) {
188 case self::INITIAL_STATE:
189 if ($token === '\'') {
191 return Success::create(['', false, self::SINGLE_QUOTED_STATE]);
192 } elseif ($token === '"') {
194 return Success::create(['', false, self::DOUBLE_QUOTED_STATE]);
195 } elseif ($token === '#') {
197 return Success::create(['', false, self::COMMENT_STATE]);
198 } elseif ($token === '$') {
200 return Success::create([$token, true, self::UNQUOTED_STATE]);
201 } else {
203 return Success::create([$token, false, self::UNQUOTED_STATE]);
204 }
205 case self::UNQUOTED_STATE:
206 if ($token === '#') {
208 return Success::create(['', false, self::COMMENT_STATE]);
209 } elseif (\ctype_space($token)) {
211 return Success::create(['', false, self::WHITESPACE_STATE]);
212 } elseif ($token === '$') {
214 return Success::create([$token, true, self::UNQUOTED_STATE]);
215 } else {
217 return Success::create([$token, false, self::UNQUOTED_STATE]);
218 }
219 case self::SINGLE_QUOTED_STATE:
220 if ($token === '\'') {
222 return Success::create(['', false, self::WHITESPACE_STATE]);
223 } else {
225 return Success::create([$token, false, self::SINGLE_QUOTED_STATE]);
226 }
227 case self::DOUBLE_QUOTED_STATE:
228 if ($token === '"') {
230 return Success::create(['', false, self::WHITESPACE_STATE]);
231 } elseif ($token === '\\') {
233 return Success::create(['', false, self::ESCAPE_SEQUENCE_STATE]);
234 } elseif ($token === '$') {
236 return Success::create([$token, true, self::DOUBLE_QUOTED_STATE]);
237 } else {
239 return Success::create([$token, false, self::DOUBLE_QUOTED_STATE]);
240 }
241 case self::ESCAPE_SEQUENCE_STATE:
242 if ($token === '"' || $token === '\\') {
244 return Success::create([$token, false, self::DOUBLE_QUOTED_STATE]);
245 } elseif ($token === '$') {
247 return Success::create([$token, false, self::DOUBLE_QUOTED_STATE]);
248 } else {
249 $first = Str::substr($token, 0, 1);
250 if (\in_array($first, ['f', 'n', 'r', 't', 'v'], true)) {
252 return Success::create([\stripcslashes('\\'.$first).Str::substr($token, 1), false, self::DOUBLE_QUOTED_STATE]);
253 } else {
255 return Error::create('an unexpected escape sequence');
256 }
257 }
258 case self::WHITESPACE_STATE:
259 if ($token === '#') {
261 return Success::create(['', false, self::COMMENT_STATE]);
262 } elseif (!\ctype_space($token)) {
264 return Error::create('unexpected whitespace');
265 } else {
267 return Success::create(['', false, self::WHITESPACE_STATE]);
268 }
269 case self::COMMENT_STATE:
271 return Success::create(['', false, self::COMMENT_STATE]);
272 default:
273 throw new \Error('Parser entered invalid state.');
274 }
275 }
