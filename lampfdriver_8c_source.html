108static int __init lampf_driver_init(void)
109{
110 if ((alloc_chrdev_region(&dev, 0, 1, DEVICE_NAME)) < 0)
111 {
112 printk(KERN_INFO "[LampF] Cannot allocate major number\n");
113 return -1;
114 }
115
116 /*Creating cdev structure*/
117 cdev_init(&c_dev, &fops);
118
119 /*Adding character device to the system*/
120 if((cdev_add(&c_dev,dev,1)) < 0)
121 {
122 printk(KERN_INFO "[LampF] Cannot add the device to the system\n");
123 unregister_chrdev_region(dev,1);
124 }
125
126 /*Creating struct class*/
127 if((dev_class = class_create(THIS_MODULE, CLASS_NAME)) == NULL)
128 {
129 printk(KERN_INFO "[LampF] Cannot create the struct class\n");
130 unregister_chrdev_region(dev,1);
131 }
132
133 /*Creating device*/
134 if((device_create(dev_class, NULL, dev, NULL, DEVICE_NAME)) == NULL)
135 {
136 printk(KERN_INFO "[LampF] Cannot create the Device\n");
137 class_destroy(dev_class);
138 unregister_chrdev_region(dev,1);
139 }
140
141 s_pGpioRegisters = (struct GpioRegisters *)ioremap(GPIO_BASE, sizeof(struct GpioRegisters));
142 // s_pGpioRegisters = (struct GpioRegisters *)ioremap_nocache(GPIO_BASE, sizeof(struct GpioRegisters));
143 SetGPIOFunction(s_pGpioRegisters, pinNum, GPIO_INPUT);
144
145 return 0;
146}
