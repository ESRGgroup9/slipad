110/**********************************************************************************************
111* @brief I2C_DEV_init()
112* @brief powers on, set gain is 16, integration time is 688,5 ms
113* @brief This field selects the integration time for each conversion.
114* @param
115**********************************************************************************************/
116bool Init_TSL2581(void)
117{
118 if(DEV_ModuleInit(ADDR_FLOAT)==1)
119 return false;
120
121 powerOnSensor();
122
123 IIC_Write(COMMAND_CMD | TIMING, INTEGRATIONTIME_688MS); //688,5 ms
124 IIC_Write(COMMAND_CMD | CONTROL, ADC_EN | CONTROL_POWERON); //Every ADC cycle generates interrupt
125 IIC_Write(COMMAND_CMD | INTERRUPT, INTR_INTER_MODE); //TEST MODE
126 IIC_Write(COMMAND_CMD | ANALOG, GAIN_1X); //GAIN = 16
127
128 return true;
129}

188/**********************************************************************************************
189* @brief calculateLux()
190* @brief Channel_0 and Channel_1 is for TSL2561_Read_Channel();
191* @param iGain - gain, where 0:1X, 1:8X, 2:16X, 3:128X
192* @param // tIntCycles - INTEG_CYCLES defined in Timing Register
193**********************************************************************************************/
194uint32_t calculateLux(uint16_t iGain)//, uint16_t tIntCycles)
195{
196 uint32_t chScale0;
197 uint32_t chScale1;
198 uint32_t channel1;
199 uint32_t channel0;
200 uint32_t temp;
201 uint32_t ratio1 = 0;
202 uint32_t ratio;
203 uint32_t lux_temp;
204 uint16_t b, m;
205
206 // // No scaling if nominal integration (148 cycles or 400 ms) is used
207 // if (tIntCycles == NOM_INTEG_CYCLE)
208 // {
209 // // chScale0 = 65536;
210 chScale0 = (1 << (CH_SCALE));
211 // }
212 // else
213 // chScale0 = (NOM_INTEG_CYCLE << CH_SCALE) / tIntCycles;
214
215 switch (iGain)
216 {
217 case 0: // 1x gain
218 chScale1 = chScale0; // No scale. Nominal setting
219 break;
220 case 1: // 8x gain
221 chScale0 = chScale0 >> 3; // Scale/multiply value by 1/8
222 chScale1 = chScale0;
223 break;
224 case 2: // 16x gain
225 chScale0 = chScale0 >> 4; // Scale/multiply value by 1/16
226 chScale1 = chScale0;
227 break;
228 case 3: // 128x gain
229 chScale1 = chScale0 / CH1GAIN128X;
230 chScale0 = chScale0 / CH0GAIN128X;
231 break;
232 }
233
234 // Read Channel for ADC
235 Read_Channel();
236 // scale the channel values
237 channel0 = (Channel_0 * chScale0) >> CH_SCALE;
238 channel1 = (Channel_1 * chScale1) >> CH_SCALE;
239 //printf(" C10= %ld C11 =%ld \n ",channel0,channel1);
240 // find the ratio of the channel values (Channel1/Channel0)
241 if (channel0 != 0)
242 ratio1 = (channel1 << (RATIO_SCALE + 1)) / channel0;
243
244 ratio = (ratio1 + 1) >> 1; // round the ratio value
245
246 if ((ratio >= 0X00) && (ratio <= K1C))
247 {
248 b = B1C;
249 m = M1C;
250 }
251 else if (ratio <= K2C)
252 {
253 b = B2C;
254 m = M2C;
255 }
256 else if (ratio <= K3C)
257 {
258 b = B3C;
259 m = M3C;
260 }
261 else if (ratio <= K4C)//276
262 {
263 b = B4C;
264 m = M4C;
265 }
266 else if (ratio > K5C)//276
267 {
268 b = B5C;
269 m = M5C;
270 }
271
272 temp = ((channel0 * b) - (channel1 * m));
273 temp += (1 << (LUX_SCALE - 1)); // round lsb (2^(LUX_SCALE-1))
274 // temp = temp + 32768;
275
276 if(channel0 == channel1 && channel0 == 4095)
277 lux_temp=MAX_LUX;
278
279 lux_temp = temp >> LUX_SCALE; // strip off fractional portion
280 return (lux_temp); // Signal I2C had no errors
281}
